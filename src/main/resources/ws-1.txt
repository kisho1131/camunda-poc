public void updateStatusCodes(List<profileCommentaryDetails> profileCommentaryDetails) {
    if (profileCommentaryDetails == null || profileCommentaryDetails.isEmpty()) {
        return;
    }

    for (profileCommentaryDetails details : profileCommentaryDetails) {
        updateBasedOnApproval(details);
        updateBasedOnStatus(details);
    }
}

private void updateBasedOnApproval(profileCommentaryDetails details) {
    if ("Y".equalsIgnoreCase(details.getIsApprovedFlag())) {
        if (Status.APPROVED.getDescription().equalsIgnoreCase(details.getStatusCd())) {
            details.setStatusCd(Status.COMPLETED.getDescription());
        } else if (Status.COMPLETED.getDescription().equalsIgnoreCase(details.getStatusCd())) {
            details.setStatusCd(Status.APPROVED.getDescription());
        } else if (Status.REJECTED.getDescription().equalsIgnoreCase(details.getStatusCd())) {
            details.setStatusCd(Status.APPROVED.getDescription());
        } else if (Status.CREATED.getDescription().equalsIgnoreCase(details.getStatusCd()) && Objects.isNull(details.getIsApprovedFlag())) {
            details.setStatusCd(Status.COMPLETED.getDescription());
        }
    }
}

private void updateBasedOnStatus(profileCommentaryDetails details) {
    if (Status.CREATED.getDescription().equalsIgnoreCase(details.getStatusCd())) {
        if (Objects.isNull(details.getCommentary())) {
            details.setStatusCd(null);
        } else if (details.getCurrInd() != null && details.getCurrInd() == 0 && Objects.nonNull(details.getCommentary())) {
            details.setStatusCd(null);
        }
    } else if (Status.REJECTED.getDescription().equalsIgnoreCase(details.getStatusCd()) && Objects.isNull(details.getIsApprovedFlag())) {
        details.setStatusCd(Status.COMPLETED.getDescription());
    }
}
